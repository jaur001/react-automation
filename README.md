## Install

```bash
npm install --save react-automation
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-automation'
import 'react-automation/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
